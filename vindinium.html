What is Vindinium?

Vindinium is an artificial intelligence programming challenge game. Each player codes some algorithms that allows their hero character to travel about a random generated map and collect mines, gaining gold and defeating other players. The objective of the game is to have the most mines so you can have the most gold and become the richest of the mighty heroes! Each player can code different actions for their character to do so they have a more likely chance to win

How The Code Works

bot.freeMines/Arrays

The basis of all of the code for the Vindinium bots is a>b>c. For example, in my code if we look at the item bot.freeMines and look at what it really does we would find that it is simply an array that holds the location for each of the mines that Vindinium randomly spawns. It would be annoying if you had to name every mine and program your bot for which once was free and closest manually. Arrays allow the bot.freeMines to hold the x,y coordinates for each of the mines on the map that are not claimed.



The task "task ==== freeMines" is the part that actually instructs the bot to do a specific action. Of course, arrays start with 0 and count each item from there, so if we look at my code and see var closestMine = bot.freeMines[0] (highlighted in yellow) we know this is the first item in the array but we don’t know if it’s actually the closest. It could be, or it could not be. So, in my code, I wrote for (var i = 0; i < bot.freeMines.length; i++) (highlighted in red) this is helpful for my code because .length is is also array, but it holds the number of objects that are in the other array. In my code bot.freeMines.length would basically be how many mines are in the spawned map.




The other part of this code that makes the action complete is (bot.findDistance(myPos, closestTavern) > bot.findDistance(myPos, bot.taverns[i])) {closestTavern = bot.taverns[i];} (shown in the picture below.) This basically instructs the computer to look back at every object in the array comparing them to each other to determine which in fact is the closest.




This whole piece of code that instructs the AI how to do a certain action is called an algorithm. In my code I use five tasks which have five different but similar algorithms. In my code, to make my bot regain health at the tavern and to attack the closest bot it has to go through the same steps but the array names are different so that the AI knows which task it is trying to accomplish.

closestOfRichest

Finding the bot that has the most gold to steal their mines has the same basic structure to find free mines. The variables, for loops and if statements all act the same but with different variable names.



However, in my global data code instead of coding to find the closests out of the mines in the array of code, this code finds the bot that is defined as the richest based on how much gold the bot owns.

var goDir;
var Promise = require('bluebird');
Bot.prototype.botBrain = function() {
    return new Promise(function(resolve, reject) {
        _this = bot;
        //////* Write your bot below Here *//////
        //////* Set `myDir` in the direction you want to go and then bot.goDir is set to myDir at the bottom *////////

        /*                                      *
         * This Code is global data!            *
         *                                      */

        // Set myDir to what you want and it will set bot.goDir to that direction at the end.  Unless it is "none"
        var myDir;

        //
        var myPos = [bot.yourBot.pos.x, bot.yourBot.pos.y];

        //
        var enemyBots = [];
        if (bot.yourBot.id != 1) enemyBots.push(bot.bot1);
        if (bot.yourBot.id != 2) enemyBots.push(bot.bot2);
        if (bot.yourBot.id != 3) enemyBots.push(bot.bot3);
        if (bot.yourBot.id != 4) enemyBots.push(bot.bot4);

        //find closest bot in closestBot
        var closestBot = enemyBots[0];
        for (var i = 0; i < enemyBots.length; i++) {
            //if you're finding a bot use .posArray for distance
            if (bot.findDistance(myPos, closestBot.posArray) > bot.findDistance(myPos, enemyBots[i].posArray)) {
                closestBot = enemyBots[i];
            }
        }
        
        // find richest enemy bot
        var richestBot = enemyBots[0];
        for (var i = 0; i < enemyBots.length; i++) {
           if (enemyBots[i].gold > richestBot.gold) {
                richestBot = enemyBots[i];
            }
        }


        /*                                      *
         * This Code Decides WHAT to do         *
         *     
         */
        var task = "none";

        if (bot.yourBot.life <= 35) {
            task = "health";
        }

        else if (bot.freeMines > bot.yourBot.mines) {
            task = "freeMines";
        }
        
        else if (richestBot.mines >= bot.yourBot.mines) {
            task = "steal";
        }
        
        else if (closestBot.life < bot.yourBot.life) {
            task = "attackClosest";

        }


        /*                                      *
         * This Code Determines HOW to do it    *
         *                                      */

        if (task === "none") {
            console.log("Going Random!");
            var rand = Math.floor(Math.random() * 4);
            var dirs = ["north", "south", "east", "west"];
            myDir = dirs[rand];
        }

        //all tasks that are coded into my bot
        else if (task === "freeMines") {
            var closestMine = bot.freeMines[0];
            for (var i = 0; i < bot.freeMines.length; i++) {
                if (bot.findDistance(myPos, closestMine) > bot.findDistance(myPos, bot.freeMines[i])) {
                    closestMine = bot.freeMines[i];
                }
            }
            console.log("Grabbing a free mine!");
            myDir = bot.findPath(myPos, closestMine);
        }

        else if (task === "health") {
            var closestTavern = bot.taverns[0];
            for (var i = 0; i < bot.taverns.length; i++) {
                if (bot.findDistance(myPos, closestTavern) > bot.findDistance(myPos, bot.taverns[i])) {
                    closestTavern = bot.taverns[i];
                }
            }
            console.log("Drinking tasty beer!");
            myDir = bot.findPath(myPos, closestTavern);
        }

        else if (task === "steal") {
            // closestOfRichest is the closest mine belonging to the richest player
            var closestOfRichest = richestBot.mines[0];
            for (var i = 0; i < richestBot.length; i++) {
                if (bot.findDistance(myPos, closestOfRichest) > bot.findDistance(myPos, richestBot[i])) {
                    closestOfRichest = richestBot[i];
                }
            }
            console.log("Taking mines hahaha!");
            myDir = bot.findPath(myPos, closestOfRichest);
        }

        else if (task === "attackClosest") {
            console.log("Attacking closest bot now!");
            myDir = bot.findPath(myPos, closestBot.posArray);
        }





        /*                                                                          *
         * This Code Sets your direction based on myDir. Change if you want.        */

        bot.goDir = myDir;

        ///////////* DON'T REMOVE ANTYTHING BELOW THIS LINE *//////////////
        resolve();
    });
}
